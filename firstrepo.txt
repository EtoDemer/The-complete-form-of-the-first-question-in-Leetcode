#include<iostream>
using namespace std;
#include<vector>

vector<int>twoSum(vector<int>& nums, int target) {
	int n = nums.size();
	vector<int>num(2, 88);
	for (int i = 0; i < n; ++i) {
		for (int j = i + 1; j < n; ++j) {
			if (nums[i] + nums[j] == target) {
				nums.push_back(i);
				nums.push_back(j);
				nums.push_back(1);
				return num;
			}
		}
	}
	return num;
}
